#include <iostream>

using namespace std;

struct user
{
	int q;
	int w;
	float e;
};                       обязательно ставим ;

Структура пишеться над мейном (Хранит в себе переменные)

int main()      
{

	user a;   а являеться объектом            создаём а которая связана со структурой для получения доступа к ней используем оператор . и выбераем имя перем со структуры, и даём ему значение
	a.q = 1;
	a.w = 10;
	a.e = 12.3;

	cout << "q " << a.q << endl; 
	cout << "w " << a.w << endl;
	cout << "e " << a.e << endl;

	user z = {12, 23, 12.5};               создаём перем я связанную со структ и задаём значение для  q, w. e
	cout << endl;
	cout << "q " << z.q << endl;
	cout << "w " << z.w << endl;
	cout << "e " << z.e << endl;
	cout << endl;
	user x = a;                        создаём перем  х и связываем с а  но меняем значение 1 перем
	x.e = 123.56;

	cout << "q " << x.q << endl;
	cout << "w " << x.w << endl;
	cout << "e " << x.e << endl;

	return 0;
}
