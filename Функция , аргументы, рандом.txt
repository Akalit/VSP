void rand(int * mas, int size)
{
	srand (time(NULL));
	for (int q =0 ; q < size ; q++)
	mas [q] = rand () % 20;
}


void showmas(int * mas, int size)
{
for (int q =0 ; q < size ; q++)
cout << " " << mas [q] << endl;
}
